1   package heaa.battleship.controller;
2   
3   import heaa.battleship.util.NavyTool;
4   import heaa.battleship.util.ShipBuilder;
5   import heaa.battleship.model.AIPlayer;
6   import heaa.battleship.model.Grid;
7   import heaa.battleship.model.HumanPlayer;
8   import heaa.battleship.model.Player;
9   import heaa.battleship.model.Position;
10  import heaa.battleship.model.Ship;
11  import java.util.List;
12  import java.util.Stack;
13  
14  public class GameInitializerController {
15  
16      private AIPlayer computer;
17      private HumanPlayer human;
18  
19      public void startGame() {
20          this.computer = new AIPlayer();
21          this.human = new HumanPlayer();
22          Grid humanGrid = new Grid(NavyTool.createNavy());
23          Grid aiGrid = new Grid(NavyTool.createNavy());
24          humanGrid.setGridSize(10);
25          aiGrid.setGridSize(10);
26          this.human.setGrid(humanGrid);
27          this.computer.setGrid(aiGrid);
28          this.setAIShipsToGrid(this.computer, ShipBuilder.createStackOfShipLengths());
29      }
30  
31      public void setAIShipsToGrid(AIPlayer computer, Stack<Integer> shipLengths) {
32          while (!shipLengths.empty()) {
33              try {
34                  addShipToGrid(
35                          computer.getGrid(),
36                          computer.determinePositionToPut(computer.getGrid()),
37                          shipLengths.peek(),
38                          computer.determineAlignment());
39                  shipLengths.pop();
40              } catch (RuntimeException e) {
41              }
42          }
43      }
44  
45      public void addShipToGrid(Grid grid, Position place, Integer shipLength, boolean horizontal) {
46          List<Position> desiredPositions = ShipBuilder.buildDesiredPositions(shipLength, horizontal, place);
47  
48          if (grid.getNavy().canShipBeAdded(desiredPositions)) {
49              Ship ship = new Ship(shipLength);
50              ship.setPositions(desiredPositions);
51              grid.getNavy().addShip(ship);
52          } else {
53              throw new RuntimeException("Ship can't be added here!");
54          }
55      }
56  
57      public Player getHumanPlayer() {
58          return this.human;
59      }
60  
61      public Player getAIPlayer() {
62          return this.computer;
63      }
64  
65  }
