1   package heaa.battleship.view;
2   
3   import heaa.battleship.model.GameSettings;
4   import javax.swing.ButtonGroup;
5   import javax.swing.JButton;
6   import javax.swing.JRadioButton;
7   import javax.swing.SwingUtilities;
8   
9   /**
10   * Luokka, joka kontrolloi pelin päävalikon mallintamista.
11   *
12   * @author heaarnio
13   */
14  public class MainMenuViewController implements Runnable {
15  
16      private MainMenuView mainMenu;
17  
18      private final String TEN = "10";
19      private final String TWENTY = "20";
20      private final String THIRTY = "30";
21      private final String FORTY = "40";
22  
23      private PositionView[][] grid;
24      private ButtonGroup gridSizeSelection;
25  
26      public MainMenuViewController() {
27          this.mainMenu = new MainMenuView();
28      }
29  
30      @Override
31      public void run() {
32          mainMenu = new MainMenuView();
33          gridSizeSelection = new ButtonGroup();
34          mainMenu.setChooseGridSizeText();
35          createSizeButtons();
36          createStartButton();
37          createExitButton();
38          MainFrame.getInstance().setPanelInView(mainMenu.getMainPanel());
39      }
40  
41      private void createSizeButtons() {
42          JRadioButton ten = new JRadioButton(TEN);
43          ten.setActionCommand(TEN);
44          ten.setSelected(true);
45          JRadioButton twenty = new JRadioButton(TWENTY);
46          twenty.setActionCommand(THIRTY);
47          JRadioButton thirty = new JRadioButton(THIRTY);
48          thirty.setActionCommand(THIRTY);
49          JRadioButton forty = new JRadioButton(FORTY);
50          forty.setActionCommand(FORTY);
51          gridSizeSelection.add(ten);
52          gridSizeSelection.add(twenty);
53          gridSizeSelection.add(thirty);
54          gridSizeSelection.add(forty);
55          this.mainMenu.setGridSizeRadioButtons(ten, twenty, thirty, forty);
56      }
57  
58      private void createStartButton() {
59          JButton start = new JButton("Start game");
60          start.addActionListener(action -> {
61              Integer size = new Integer(gridSizeSelection.getSelection().getActionCommand());
62              GameSettings.getInstance().setGridSize(size);
63              SwingUtilities.invokeLater(new ShipDeploymentViewController());
64          });
65          this.mainMenu.setStartButton(start);
66      }
67  
68      private void createExitButton() {
69          JButton exit = new JButton("Exit");
70          exit.addActionListener(action -> {
71              System.exit(0);
72          });
73          this.mainMenu.setExitButton(exit);
74      }
75  
76  }
