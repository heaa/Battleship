17  public class GameEndViewController implements Runnable {
18  
19      private GameEndView gameEnd;
20  
21      @Override
22      public void run() {
23          this.gameEnd = new GameEndView();
24          gameStatistics();
25          createMainMenuButton();
26          MainFrame.getInstance().setPanelInView(this.gameEnd.getMainPanel());
27      }
28  
29      private void gameStatistics() {
30          AIPlayer computer = GameSettings.getInstance().getAiPlayer();
31          HumanPlayer human = GameSettings.getInstance().getHumanPlayer();
32          int computerMissed = computer.getGrid().getMissedPositions().size();
33          int computerHits = computer.getGrid().getDestroyedShipPositions().size();
34  
35          JPanel stats = new JPanel();
36          stats.setLayout(new GridBagLayout());
37  
38          stats.add(createWinnerText());
39          stats.add(createPointText());
40          this.gameEnd.setGameStatistics(stats);
41      }
42  
43      private JLabel createWinnerText() {
44          JLabel winnerText = new JLabel();
45          if (humanWon()) {
46              winnerText.setText("Game ended. You won!");
47          } else {
48              winnerText.setText("Game ended. Computer won!");
49          }
50          return winnerText;
51      }
52  
53      private JLabel createPointText() {
54          JLabel pointText = new JLabel();
55          if (humanWon()) {
56              pointText.setText("You collected " + calculateHumanPoints(GameSettings.getInstance().getAiPlayer()) + " points.");
57          } else {
58              pointText.setText("Computer collected " + calculateComputerPoints(GameSettings.getInstance().getHumanPlayer()) + " points.");
59          }
60          return pointText;
61      }
62  
63      private boolean humanWon() {
64          if (GameSettings.getInstance().getHumanPlayer().hasShips()) {
65              return true;
66          }
67          return false;
68      }
69  
70      private long calculateHumanPoints(AIPlayer computer) {
71          int squares = (int) Math.pow(GameSettings.getInstance().getGridSize(), 2);
72          int computerShooted = computer.getGrid().getShootedPositions().size();
73          return calculatePoints(squares, computerShooted);
74      }
75  
76      private long calculateComputerPoints(HumanPlayer human) {
77          int squares = (int) Math.pow(GameSettings.getInstance().getGridSize(), 2);
78          int humanShooted = human.getGrid().getShootedPositions().size();
79          return calculatePoints(squares, humanShooted);
80      }
81  
82      private long calculatePoints(int squares, int shooted) {
83          return Math.round((squares / shooted) * 1000);
84      }
85  
86      private void createMainMenuButton() {
87          JButton mainMenu = new JButton("To main menu");
88          mainMenu.addActionListener(action -> {
89              SwingUtilities.invokeLater(new MainMenuViewController());
90          });
91          gameEnd.setMainMenuButton(mainMenu);
92      }
93  }
